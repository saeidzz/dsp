489     public  static void bubbleSort(int[] arr,String[] qoute,int k) {  
490         int n = arr.length;  
491         int temp = 0;
492         String tmpstr="";
493          for(int i=0; i < n; i++){  
494                  for(int j=1; j < (n-i); j++){  
495                           if(arr[j-1] > arr[j]){  
496                                  //swap elements  
497                                  temp = arr[j-1]; 
498                                  tmpstr=qoute[j-1];
499                                  
500                                  arr[j-1] = arr[j]; 
501                                  qoute[j-1]=qoute[j];
502                                  
503                                  arr[j] = temp;  
504                                  qoute[j]=tmpstr;
505                          }  
506                           
507                  }  
508          }
509          for (int i = arr.length-1; i > arr.length-k; i--) {
510              System.out.println(qoute[i]+"  and the vote is :"+arr[i]);
511        }
512   
513     }  
514     public static void fileCleaner() throws IOException{
